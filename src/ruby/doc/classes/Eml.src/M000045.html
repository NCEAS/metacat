  def largest_data_table
    if(data_tables.length == 1)
      return data_tables[0]
    else
      size = 0
      largest = nil
      data_tables.each do |data_table|
        if(data_table.size > size)
          size = data_table.size
          largest = data_table
        end
      end
      largest
    end
  end
